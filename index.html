a canvas-based biological charting library that makes it easy to create charts of alignment and assembly data
Get Started Quickly
// Get Canvas and Create Chart
var canvas = document.getElementById(canvasName);

// Create Chart with canvas, width
chart = new Scribl(canvas, 500);

// Add Gene with position, length, orientation
gene1 = chart.addGene( 5, 750 , '+');

// Draw Chart
chart.draw();
